### 🔧 Enable Pac-Man: Add this workflow to your repo

```yaml
# .github/workflows/pacman.yml
name: Generate Pac-Man animation

on:
  schedule:
    - cron: "0 */12 * * *"   # runs every 12 hours
  workflow_dispatch:
  push:
    branches:
      - main

jobs:
  generate:
    permissions:
      contents: write
    runs-on: ubuntu-latest
    timeout-minutes: 10

    steps:
      - name: Generate Pac-Man animation 🟡
        uses: Platane/snk/svg-only@v3
        with:
          github_user_name: ${{ github.repository_owner }}
          outputs: |
            dist/pacman-contribution-grid.svg
            dist/pacman-contribution-grid-dark.svg?palette=github-dark
          svg_out_path: dist/pacman.svg
          # Use pacman theme
          color_snake: "#FFCC00"
          color_dots: "#a78bfa,#7c3aed"
          color_empty: "#161b22"
          color_board: "#0d1117"

      - name: Push to output branch
        uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```
